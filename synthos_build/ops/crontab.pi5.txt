# SYNTHOS RETAIL — updated 2026-04-20
# All times America/New_York (Pi5 16GB system clock)
# Market daemon (intel): news, sentiment, macro, validator, screener, price poller (9:10-16:00)
# Trade daemon (action): continuous trader during market hours 9:30-16:00
# Cron handles: overnight news, boot, heartbeat, backup, maintenance

# Boot services

# ── MARKET DAEMON — enrichment pipeline 9:10-16:00 weekdays ──
# Kicks at 9:10 to warm up pre-open. Runs news + sentiment + macro + validator +
# screener + price poller on a 30-min enrichment cadence. Does NOT run the
# trader — that's the trade daemon below.
10 9 * * 1-5  cd /home/pi516gb/synthos/synthos_build && /usr/bin/python3 src/retail_market_daemon.py >> logs/market_daemon.log 2>&1

# ── TRADE DAEMON — continuous trader 9:30-16:00 weekdays ──
# Phase 1 of TRADER_RESTRUCTURE_PLAN: trader runs on its own continuous loop
# (30s cycle) instead of every-30-min from market_daemon. Starts at 9:25 so
# it's ready at market open; the daemon itself waits for 9:30 ET before
# firing its first dispatch.
25 9 * * 1-5  cd /home/pi516gb/synthos/synthos_build && /usr/bin/python3 src/retail_trade_daemon.py >> logs/trade_daemon.log 2>&1

# ── OVERNIGHT NEWS — off-hours only (daemon handles 9-4) ──
5 0-8,16-23 * * 1-5 /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/retail_scheduler.py --session overnight >> /home/pi516gb/synthos/synthos_build/logs/scheduler.log 2>&1

# ── Node heartbeat — every minute ──
* * * * * /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/node_heartbeat.py /home/pi516gb/synthos/synthos_build/user/.env >> /home/pi516gb/synthos/synthos_build/logs/heartbeat.log 2>&1

# ── Retail heartbeat — every 20 min (agent status to monitor) ──
21 * * * 1-5 /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/retail_heartbeat.py --session trade >> /home/pi516gb/synthos/synthos_build/logs/heartbeat.log 2>&1

# ── Sunday evening prep — build Monday context ──
0 20 * * 0  /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/retail_scheduler.py --session prep >> /home/pi516gb/synthos/synthos_build/logs/scheduler.log 2>&1

# ── Saturday maintenance ──
55 3 * * 6  /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/retail_shutdown.py
0  4 * * 6  sudo reboot

# ── BACKUP — nightly 1:30am ET to pi4b staging ──
30 1 * * *  /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/retail_backup.py >> /home/pi516gb/synthos/synthos_build/logs/backup.log 2>&1

# ── NIGHTLY TEMPLATE REBUILD ──
0 3 * * * /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/rebuild_default_template.py >> /home/pi516gb/synthos/synthos_build/logs/template_rebuild.log 2>&1
# ── LOG ROTATION — weekly Sunday midnight ──
0 0 * * 0 /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/rotate_logs.py >> /home/pi516gb/synthos/synthos_build/logs/rotation.log 2>&1

# ── WEEKEND OVERNIGHT SCHEDULER — hourly, :05 past ──
# Matches the weekday :05 pattern but runs 24 hours/day Sat+Sun
# since markets are closed and market_daemon doesn't run then.
# :05 deliberately chosen to dodge :00 (reboot/rebuild) and :30 (backup).
5 * * * 0,6 /usr/bin/python3 /home/pi516gb/synthos/synthos_build/src/retail_scheduler.py --session overnight >> /home/pi516gb/synthos/synthos_build/logs/scheduler.log 2>&1
